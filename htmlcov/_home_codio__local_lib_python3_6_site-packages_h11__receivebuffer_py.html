7# Operations we want to support: 

8# - find next \r\n or \r\n\r\n (\n or \n\n are also acceptable), 

9# or wait until there is one 

10# - read at-most-N bytes 

11# Goals: 

12# - on average, do this fast 

13# - worst case, do this in O(n) where n is the number of bytes processed 

14# Plan: 

15# - store bytearray, offset, how far we've searched for a separator token 

16# - use the how-far-we've-searched data to avoid rescanning 

17# - while doing a stream of uninterrupted processing, advance offset instead 

18# of constantly copying 

19# WARNING: 

20# - I haven't benchmarked or profiled any of this yet. 

21# 

22# Note that starting in Python 3.4, deleting the initial n bytes from a 

23# bytearray is amortized O(n), thanks to some excellent work by Antoine 

24# Martin: 

25# 

26# https://bugs.python.org/issue19087 

27# 

28# This means that if we only supported 3.4+, we could get rid of the code here 

29# involving self._start and self.compress, because it's doing exactly the same 

30# thing that bytearray now does internally. 

31# 

32# BUT unfortunately, we still support 2.7, and reading short segments out of a 

33# long buffer MUST be O(bytes read) to avoid DoS issues, so we can't actually 

34# delete this code. Yet: 

35# 

36# https://pythonclock.org/ 

37# 

38# (Two things to double-check first though: make sure PyPy also has the 

39# optimization, and benchmark to make sure it's a win, since we do have a 

40# slightly clever thing where we delay calling compress() until we've 

41# processed a whole event, which could in theory be slightly more efficient 

42# than the internal bytearray support.) 

43blank_line_regex = re.compile(b"\n\r?\n", re.MULTILINE)

86 def maybe_extract_next_line(self): 

87 """ 

88 Extract the first line, if it is completed in the buffer. 

89 """ 

90 # Only search in buffer space that we've not already looked at. 

91 search_start_index = max(0, self._next_line_search - 1) 

92 partial_idx = self._data.find(b"\r\n", search_start_index) 

93 

94 if partial_idx == -1: 

95 self._next_line_search = len(self._data) 

96 return None 

97 

98 # + 2 is to compensate len(b"\r\n") 

99 idx = partial_idx + 2 

100 

101 return self._extract(idx)

103 def maybe_extract_lines(self): 

104 """ 

105 Extract everything up to the first blank line, and return a list of lines. 

106 """ 

107 # Handle the case where we have an immediate empty line. 

108 if self._data[:1] == b"\n": 

109 self._extract(1) 

110 return [] 

111 

112 if self._data[:2] == b"\r\n": 

113 self._extract(2) 

114 return [] 

115 

116 # Only search in buffer space that we've not already looked at. 

117 match = blank_line_regex.search(self._data, self._multiple_lines_search) 

118 if match is None: 

119 self._multiple_lines_search = max(0, len(self._data) - 2) 

120 return None 

121 

122 # Truncate the buffer and return it. 

123 idx = match.span(0)[-1] 

124 out = self._extract(idx) 

125 lines = out.split(b"\n") 

126 

127 for line in lines: 

128 if line.endswith(b"\r"): 

129 del line[-1] 

130 

131 assert lines[-2] == lines[-1] == b"" 

132 

133 del lines[-2:] 

134 

135 return lines

137 # In theory we should wait until `\r\n` before starting to validate 

138 # incoming data. However it's interesting to detect (very) invalid data 

139 # early given they might not even contain `\r\n` at all (hence only 

140 # timeout will get rid of them). 

141 # This is not a 100% effective detection but more of a cheap sanity check 

142 # allowing for early abort in some useful cases. 

143 # This is especially interesting when peer is messing up with HTTPS and 

144 # sent us a TLS stream where we were expecting plain HTTP given all 

145 # versions of TLS so far start handshake with a 0x16 message type code. 

146 def is_next_line_obviously_invalid_request_line(self): 

147 try: 

148 # HTTP header line must not contain non-printable characters 

149 # and should not start with a space 

150 return self._data[0] < 0x21 

151 except IndexError: 

152 return False
