53# Some convenience utilities for working with HTTP headers 

54def normed_header_dict(h11_headers: Headers) -> Dict[bytes, bytes]: 

55 # This mangles Set-Cookie headers. But it happens that we don't care about 

56 # any of those, so it's OK. For every other HTTP header, if there are 

57 # multiple instances then you're allowed to join them together with 

58 # commas. 

59 name_to_values: Dict[bytes, List[bytes]] = {} 

60 for name, value in h11_headers: 

61 name_to_values.setdefault(name, []).append(value) 

62 name_to_normed_value = {} 

63 for name, values in name_to_values.items(): 

64 name_to_normed_value[name] = b", ".join(values) 

65 return name_to_normed_value

68# We use this for parsing the proposed protocol list, and for parsing the 

69# proposed and accepted extension lists. For the proposed protocol list it's 

70# fine, because the ABNF is just 1#token. But for the extension lists, it's 

71# wrong, because those can contain quoted strings, which can in turn contain 

72# commas. XX FIXME 

73def split_comma_header(value: bytes) -> List[str]: 

74 return [piece.decode("ascii").strip() for piece in value.split(b",")]

83def generate_accept_token(token: bytes) -> bytes: 

84 accept_token = token + ACCEPT_GUID
